1. řádný termín

$\frac{8}{5} \cdot \left(\frac{5}{6} \cdot \frac{7}{10} - \frac{5}{6}\right)$


$ =\frac{8}{5} \cdot \left(\frac{\cancel{\color{red}5}}{6} \cdot \frac{7}{\cancel{\color{red}10}} - \frac{5}{6}\right) =\frac{8}{5} \cdot \left(\frac{1}{6} \cdot \frac{7}{2} - \frac{5}{6}\right) $ $ =\frac{8}{5} \cdot \left(\frac{7}{12}- \frac{5}{6}\right) =\frac{8}{5} \cdot \left(\frac{7\,-\,5\,\cdot\,2}{12}\right) =\frac{8}{5} \cdot \frac{-3}{12} =\frac{8}{5} \cdot \frac{\cancel{\color{red}-3}}{\cancel{\color{\red}12}} $ $ =\frac{8}{5} \cdot \frac{-1}{4} =\frac{\cancel{\color{red}8}}{5} \cdot \frac{-1}{\cancel{\color{red}4}} =\frac{2}{5} \cdot \frac{-1}{1} =-\frac{2}{5} $

$$\frac{\left(\frac{4}{5} - \frac{2}{3}\right) \cdot \frac{5}{8}}{\frac{2}{3}}$$


$$ =\frac{\left(\frac{4\,\cdot\,3\,-\,2\,\cdot\,5}{15}\right) \cdot \frac{5}{8}}{\frac{2}{3}} =\frac{\left(\frac{12\,-\,10}{15}\right) \cdot \frac{5}{8}}{\frac{2}{3}} $$ $$ =\frac{\frac{\cancel{\color{red}2}}{\cancel{\color{green}15}} \cdot \frac{\cancel{\color{green}5}}{\cancel{\color{red}8}}}{\frac{2}{3}} =\frac{\frac{1}{3} \cdot \frac{1}{4}}{\frac{2}{3}} =\frac{\frac{1}{12}}{\frac{2}{3}} $$ $ =\frac{1}{12} \cdot \frac{3}{2} =\frac{1}{\cancel{\color{red}12}} \cdot \frac{\cancel{\color{red}3}}{2} =\frac{1}{4} \cdot \frac{1}{2} =\frac{1}{8} $

2. řádný termín

$\frac{7}{5} \cdot \frac{3}{8} \cdot \frac{10}{21} + \frac{3}{10}$


$ =\frac{\cancel{\color{blue}7}}{\cancel{\color{red}5}} \cdot \frac{3}{8} \cdot \frac{\cancel{\color{red}10}}{\cancel{\color{blue}21}} + \frac{3}{10} =\frac{1}{1} \cdot \frac{\cancel{\color{green}3}}{\cancel{\color{brown}8}} \cdot \frac{\cancel{\color{brown}2}}{\cancel{\color{green}3}} + \frac{3}{10} $ $ =\frac{1}{1} \cdot \frac{1}{4} \cdot \frac{1}{1} + \frac{3}{10} =\frac{1}{4} + \frac{3}{10} =\frac{1\,\cdot\,5\,+\,3\,\cdot\,2}{20} =\frac{5\,+\,6}{20} =\frac{11}{20} $

$$\frac{\frac{1}{4} - \frac{5}{8}}{3 \cdot \frac{5}{12}}$$


$$ =\frac{\frac{1\,\cdot\,2\,-\,5}{8}}{\frac{\cancel{\color{red}3}}{1} \cdot \frac{5}{\cancel{\color{red}12}}} =\frac{\frac{2\,-\,5}{8}}{\frac{1}{1} \cdot \frac{5}{4}} $$ $ =\frac{-3}{8} \cdot \frac{4}{5} =\frac{-3}{\cancel{\color{blue}8}} \cdot \frac{\cancel{\color{blue}4}}{5} =\frac{-3}{2} \cdot \frac{1}{5} =-\frac{3}{10} $

1. náhradní termín

$\frac{1}{3} \cdot \left(5 - \frac{13}{5}\right) : 20$


$ =\frac{1}{3} \cdot \left(\frac{25\,-\,13}{5}\right) : 20 =\frac{1}{3} \cdot \frac{12}{5} \cdot \frac{1}{20} =\frac{1}{\cancel{\color{red}3}} \cdot \frac{\cancel{\color{red}12}}{5} \cdot \frac{1}{20} $ $ =\frac{1}{1} \cdot \frac{4}{5} \cdot \frac{1}{20} =\frac{1}{1} \cdot \frac{\cancel{\color{blue}4}}{5} \cdot \frac{1}{\cancel{\color{blue}20}} =\frac{1}{1} \cdot \frac{1}{5} \cdot \frac{1}{5} =\frac{1}{25} $

$$\frac{\frac{2}{3} - \frac{3}{2}}{\frac{2}{3} : \frac{3}{2}}$$


$$ =\frac{\frac{2\,\cdot\,2\,-\,3\,\cdot\,3}{6}}{\frac{2}{3} \cdot \frac{2}{3}} =\frac{\frac{4\,-\,9}{6}}{\frac{4}{9}} =\frac{\frac{-5}{6}}{\frac{4}{9}} $$ $ =\frac{-5}{6} \cdot \frac{9}{4} =\frac{-5}{\cancel{\color{red}6}} \cdot \frac{\cancel{\color{red}9}}{4} =\frac{-5}{2} \cdot \frac{3}{4} =-\frac{15}{8} $

2. náhradní termín

$\left(\frac{12}{5} \cdot \frac{3}{20} - \frac{3}{20}\right) : \frac{7}{25}$


$ =\left(\frac{\cancel{\color{red}12}}{5} \cdot \frac{3}{\cancel{\color{red}20}} - \frac{3}{20}\right) : \frac{7}{25} =\left(\frac{3}{5} \cdot \frac{3}{5} - \frac{3}{20}\right) : \frac{7}{25} $ $ =\left(\frac{9}{25} - \frac{3}{20}\right) : \frac{7}{25} =\frac{9\,\cdot\,4\,-\,3\,\cdot\,5}{100} : \frac{7}{25} $

$ =\frac{36\,-\,15}{100} \cdot \frac{25}{7} =\frac{21}{100} \cdot \frac{25}{7} $ $ =\frac{\cancel{\color{blue}21}}{\cancel{\color{red}100}} \cdot \frac{\cancel{\color{red}25}}{\cancel{\color{blue}7}} =\frac{3}{4} \cdot \frac{1}{1} =\frac{3}{4} $

$$\frac{12}{2 + \frac{2}{3}} \cdot \frac{2 \cdot \frac{2}{3}}{18}$$


$$ =\frac{\frac{12}{1}}{\frac{2\,\cdot\,3\,+\,2}{3}} \cdot \frac{\frac{2}{1} \cdot \frac{2}{3}}{\frac{18}{1}} =\frac{\frac{12}{1}}{\frac{8}{3}} \cdot \frac{\frac{4}{3}}{\frac{18}{1}} $$ $ =\frac{12}{1} \cdot \frac{3}{8} \cdot \frac{4}{3} \cdot \frac{1}{18} =\frac{\cancel{\color{red}12}}{1} \cdot \frac{\cancel{\color{green}3}}{\cancel{\color{blue}8}} \cdot \frac{\cancel{\color{blue}4}}{\cancel{\color{green}3}} \cdot \frac{1}{\cancel{\color{red}18}} =\frac{2}{1} \cdot \frac{1}{2} \cdot \frac{1}{1} \cdot \frac{1}{3} $ $ =\frac{\cancel{\color{brown}2}}{1} \cdot \frac{1}{\cancel{\color{brown}2}} \cdot \frac{1}{1} \cdot \frac{1}{3} =\frac{1}{1} \cdot \frac{1}{1} \cdot \frac{1}{1} \cdot \frac{1}{3} =\frac{1}{3} $
