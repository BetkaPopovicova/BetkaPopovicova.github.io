1. řádný termín

$\left(\frac{5}{8} \cdot \frac{10}{9} - \frac{4}{9}\right) : \left(8 \cdot \frac{1}{6}\right)$


$ =\left(\frac{5}{\cancel{\color{red}8}} \cdot \frac{\cancel{\color{red}10}}{9} - \frac{4}{9}\right) : \left(\frac{8}{1} \cdot \frac{1}{6}\right) =\left(\frac{5}{4} \cdot \frac{5}{9} - \frac{4}{9}\right) : \left(\frac{\cancel{\color{blue}8}}{1} \cdot \frac{1}{\cancel{\color{blue}6}}\right) $ $ =\left(\frac{25}{36} - \frac{4}{9}\right) : \left(\frac{4}{1} \cdot \frac{1}{3} \right) =\frac{25\,-\,4\,\cdot\,4}{36} : \frac{4}{3} =\frac{25\,-\,16}{36} \cdot \frac{3}{4} $ $ =\frac{\cancel{\color{green}9}}{\cancel{\color{green}36}} \cdot \frac{3}{4} =\frac{1}{4} \cdot \frac{3}{4} =\frac{3}{16} $

$$\frac{2 - \frac{13}{10}}{\frac{5}{3} - \frac{1}{2}}$$


$$ =\frac{\frac{2\,\cdot\,10\,-\,13}{10}}{\frac{5\,\cdot\,2\,-\,1\,\cdot\,3}{6}} =\frac{\frac{20\,-\,13}{10}}{\frac{10\,-\,3}{6}} =\frac{\frac{7}{10}}{\frac{7}{6}} $$ $ =\frac{7}{10} \cdot \frac{6}{7} =\frac{\cancel{\color{red}7}}{\cancel{\color{blue}10}} \cdot \frac{\cancel{\color{blue}6}}{\cancel{\color{red}7}} =\frac{1}{5} \cdot \frac{3}{1} =\frac{3}{5} $

2. řádný termín

$$\frac{2 - \frac{4}{7}}{3 - \frac{13}{21}}$$


$$ =\frac{\frac{2\,\cdot\,7\,-\,4}{7}}{\frac{3\,\cdot\,21\,-\,13}{21}} =\frac{\frac{14\,-\,4}{7}}{\frac{63\,-\,13}{21}} =\frac{\frac{10}{7}}{\frac{50}{21}} $$ $ =\frac{10}{7} \cdot \frac{21}{50} =\frac{\cancel{\color{red}10}}{\cancel{\color{green}7}} \cdot \frac{\cancel{\color{green}21}}{\cancel{\color{red}50}} =\frac{1}{1} \cdot \frac{3}{5} =\frac{3}{5} $

$\left(\frac{3}{8} - \frac{2}{5}\right) \cdot 5 - \frac{3}{4}$


$ =\left(\frac{3\,\cdot\,5\,-\,2\,\cdot\,8}{40}\right) \cdot 5 - \frac{3}{4} =\left(\frac{15\,-\,16}{40} \right) \cdot 5 - \frac{3}{4} $ $ =\frac{-1}{40} \cdot \frac{5}{1} - \frac{3}{4} =\frac{-1}{\cancel{\color{red}40}} \cdot \frac{\cancel{\color{red}5}}{1} - \frac{3}{4} =\frac{-1}{8} \cdot \frac{1}{1} - \frac{3}{4} $ $ =\frac{-1}{8} - \frac{3}{4} =\frac{-1\,-\,3\,\cdot\,2}{8} =\frac{-1\,-\,6}{8} =-\frac{7}{8} $

1. náhradní termín

$\left(\frac{3}{4} + \frac{13}{6}\right) \cdot \left(\frac{2}{5} - 1\right)$


$ =\left(\frac{3\,\cdot\,3\,+\,13\,\cdot\,2}{12}\right) \cdot \left(\frac{2\,-\,1\,\cdot\,5}{5}\right) =\left(\frac{9\,+\,26}{12}\right) \cdot \left(\frac{2\,-\,5}{5}\right) $ $ =\frac{35}{12} \cdot \frac{-3}{5} =\frac{\cancel{\color{red}35}}{\cancel{\color{blue}12}} \cdot \frac{\cancel{\color{blue}-3}}{\cancel{\color{red}5}} $ $ =\frac{7}{4} \cdot \frac{-1}{1} =-\frac{7}{4} $

$$\frac{\frac{3}{5} \cdot 2 - 4 \cdot \frac{2}{7}}{2}$$


$$ =\frac{\frac{3}{5} \cdot \frac{2}{1} - \frac{4}{1} \cdot \frac{2}{7}}{\frac{2}{1}} =\frac{\frac{6}{5} - \frac{8}{7}}{\frac{2}{1}} =\frac{\frac{6\,\cdot\,7\,-\,8\,\cdot\,5}{35}}{\frac{2}{1}} $$ $$ =\frac{\frac{42\,-\,40}{35}}{\frac{2}{1}} =\frac{\frac{2}{35}}{\frac{2}{1}} $$ $ =\frac{2}{35} \cdot \frac{1}{2} =\frac{\cancel{\color{red}2}}{35} \cdot \frac{1}{\cancel{\color{red}2}} =\frac{1}{35} \cdot \frac{1}{1} =\frac{1}{35} $

2. náhradní termín

$\left(\frac{5}{8} - \frac{5}{12}\right) \cdot 4 - 2 \cdot \left(\frac{3}{4} - \frac{2}{3}\right)$


$ =\left(\frac{5\,\cdot\,3\,-\,5\,\cdot\,2}{24}\right) \cdot 4 - 2 \cdot \left(\frac{3\,\cdot\,3\,-\,2\,\cdot\,4}{12}\right) $ $ =\left(\frac{15\,-\,10}{24}\right) \cdot 4 - 2 \cdot \left(\frac{9\,-\,8}{12}\right) $ $ =\frac{5}{24} \cdot \frac{4}{1} - \frac{2}{1} \cdot \frac{1}{12} =\frac{5}{\cancel{\color{red}24}} \cdot \frac{\cancel{\color{red}4}}{1} - \frac{\cancel{\color{red}2}}{1} \cdot \frac{1}{\cancel{\color{red}12}} $ $ =\frac{5}{6} \cdot \frac{1}{1} - \frac{1}{1} \cdot \frac{1}{6} =\frac{5}{6} - \frac{1}{6} =\frac{4}{6} =\frac{\cancel{\color{blue}4}}{\cancel{\color{blue}6}} =\frac{2}{3} $

$$\frac{\left(\frac{27}{10} \cdot \frac{5}{9} - 4\right) : 3}{5}$$


$$ =\frac{\left(\frac{\cancel{\color{red}27}}{\cancel{\color{green}10}} \cdot \frac{\cancel{\color{green}5}}{\cancel{\color{red}9}} - 4\right) : 3}{5} =\frac{\left(\frac{3}{2} \cdot \frac{1}{1} - 4\right) : 3}{5} $$ $$ =\frac{\left(\frac{3\,-\,4\,\cdot\,2}{2}\right) : 3}{5} =\frac{\left(\frac{3\,-\,8}{2}\right) : 3}{5} =\frac{\frac{-5}{2} : \frac{3}{1}}{5} =\frac{\frac{-5}{2} \cdot \frac{1}{3}}{\frac{5}{1}} $$ $ =\frac{-5}{6} \cdot \frac{1}{5} =\frac{\cancel{\color{red}-5}}{6} \cdot \frac{1}{\cancel{\color{red}5}} =\frac{-1}{6} \cdot \frac{1}{1} =-\frac{1}{6} $

Ilustrační test

$\frac{1}{2} + \frac{8}{5} \cdot \left(\frac{3}{8} - \frac{1}{6}\right)$


$ =\frac{1}{2} + \frac{8}{5} \cdot \left(\frac{3\,\cdot\,3\,-\,1\,\cdot\,4}{24}\right) =\frac{1}{2} + \frac{8}{5} \cdot \left(\frac{9\,-\,4}{24}\right) $ $ =\frac{1}{2} + \frac{8}{5} \cdot \frac{5}{24} =\frac{1}{2} + \frac{\cancel{\color{blue}8}}{\cancel{\color{red}5}} \cdot \frac{\cancel{\color{red}5}}{\cancel{\color{blue}24}} $ $ =\frac{1}{2} + \frac{1}{1} \cdot \frac{1}{3} =\frac{1}{2} + \frac{1}{3} =\frac{1\,\cdot\,3\,+\,1\,\cdot\,2}{6} $ $ =\frac{3\,+\,2}{6} =\frac{5}{6} $

$$\frac{\frac{7}{4} - 4}{7 - \frac{4}{7}}$$


$$ =\frac{\frac{7}{4} - \frac{4}{1}}{\frac{7}{1} - \frac{4}{7}} =\frac{\frac{7\,-\,4\,\cdot\,4}{4}}{\frac{7\,\cdot\,7\,-\,4}{7}} $$ $$ =\frac{\frac{7\,-\,16}{4}}{\frac{49\,-\,4}{7}} =\frac{\frac{-9}{4}}{\frac{45}{7}} $$ $ =\frac{-9}{4} \cdot \frac{7}{45} =\frac{\cancel{\color{red}-9}}{4} \cdot \frac{7}{\cancel{\color{red}45}} =\frac{-1}{4} \cdot \frac{7}{5} =\frac{-7}{20} $
